# 一、防抖

# 应用场景：

  • 支付

连续点击，间歇时间小于 s，只触发一次

function debounce(fn, s, immediate) {
  let timer = null
  let result
  return function(...val) {
    clearTimeout(timer)
    if (immediate) {
      if (!timer) {
        result = fn.apply(this, val)
      }
      timer = setTimeout(() => {
        timer = null
      }, s * 1000)
    } else {
      timer = setTimeout(() => {
        fn.apply(this, val)
      }, s * 1000)
    }
    return result
  }
}

# 应用场景：

function throttle(fn,time){
  let last = 0
  let result
  let timer
  return function (...args){
    clearTimeout(timer)
    if(!last || last + time*1000 < +new Date()) {
      last = +new Date()
      result = fn.apply(this,args)
    } else {
      timer = setTimeout(() => {
        result = fn.apply(this,args)
      },time * 1000)
    }
    return result
  }
}

# 五、 深拷贝--shallowCopy、deepCopy

注： symbol类型的key就无法拷贝

# 应用场景：

  • 很多实际场景，避免原始数据受到更改
function shallowCopy(obj){
  if(typeof obj !== 'object') return
  let newObj = obj instanceof Array?[]:{}
  for(var key in obj) {
    if(obj.hasOwnProperty(key)) {
      if(typeof obj[key] === 'object' && obj[key] !== null) {
        newObj[key] = arguments.callee(obj[key])
      } else {
        newObj[key] = obj[key]
      }
    }
  }
  return newObj
}

function flatten(arr){
  while(arr.some((item) => Array.isArray(item))){

# 九、相等 ===

# 1. -0 与 +0

目标： -0 === +0

实际： -0 !== +0

// 表现1
console.log(+0 === -0); // true

// 表现2
(-0).toString() // '0'
(+0).toString() // '0'

// 表现3
-0 < +0 // false
+0 < -0 // false

// 即便如此，两者依然是不同的：

1 / +0 // Infinity
1 / -0 // -Infinity

1 / +0 === 1 / -0 // false

解释：

这是因为 JavaScript 采用了IEEE_754 浮点数表示法(几乎所有现代编程语言所采用)，这是一种二进制表示法，按照这个标准，最高位是符号位(0 代表正，1 代表负)，剩下的用于表示大小。而对于零这个边界值 ，1000(-0) 和 0000(0)都是表示 0 ，这才有了正负零的区别。

怎么得到-0

Math.round(-0.1) // -0

eq

function eq(a, b){
    if (a === b) return a !== 0 || 1 / a === 1 / b;
    return false;
}

console.log(eq(0, 0)) // true
console.log(eq(0, -0)) // false

# 第一版

function sub_curry (fn) {
  // 拿到参数
  let args = [].slice.call(arguments,1)
  return function () {
    return fn.apply(this,[].concat(args,...arguments)) // 传入多层参数
  }
}

function curry(fn,length){
  length = length || fn.length
  let _curry = arguments.callee
  return function (){
    if(arguments.length<length) {
      return _curry(sub_curry(fn,...arguments),length-arguments.length)
    } else {
      return fn.apply(this,[...arguments])
    }
  }
}

# 十二、函数组合

# 应用场景：

  • 很常用，真的，实际开发，多个函数，处理上一个函数的执行结果
function a (x) {
  return x + ' yes'
}
function b (x) {
  return 'hello ' + x
}
function c (x) {
  return x.toUpperCase()
}
var result = c(a(b('ccc'))) // 'HELLO CCC YES'

目标

var foo = compose(c,a,b)
foo('ccc')// 'HELLO CCC YES'

function compose (){
  let args = arguments
  let length = arguments.length
  return function (){
    let index = length
    var last
    while(index--){
      last =  args[index].apply(this,last?[last]:arguments)
    }
    return last
  }
}

# 特别补充

# 1. void 0

由于js中， undefined ，并不是保留字段，所以一般情况下，判断是否是 undefined 使用 void 0

void 0 === undefined

总结

  1. 通过采用 void 0 取 undefined 比采用字面上的 undefined 更靠谱更安全，应该优先采用 void 0 这种方式。

  2. 填充 <a> 的 href 确保点击时不会产生页面跳转; 填充 <image> 的 src，确保不会向服务器发出垃圾请求。
